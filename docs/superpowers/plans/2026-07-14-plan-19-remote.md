**Architecture:** GUI-first. User connects via session manager UI or `zerminal connect ssh://user@host`. Client SSHes to remote, probes for server, installs if missing, connects via SSH-forwarded local socket.

```rust
pub async fn connect_ssh(target: &str) -> Result<MuxDomain> {
    // 1. Open SSH session
    let session = ssh::connect(target).await?;
    
    // 2. Probe for zerminal-server
    let server_path = probe_remote_server(&session).await?;
    
    // 3. If not found, install
    let server_path = if server_path.is_some() {
        server_path.unwrap()
    } else {
        install_remote_server(&session).await?
    };
    
    // 4. Spawn remote server
    session.exec(format!("{} --daemonize", server_path)).await?;
    
    // 5. Forward local socket through SSH channel
    let channel = session.channel_forward("unix:/tmp/zerminal-mux.sock").await?;
    
    // 6. Create MuxDomain with Ssh transport
    Ok(MuxDomain::new(MuxTransport::Ssh { channel }))
}
```

### Task 2: Implement remote server auto-install

**Files:**
- Create: `crates/mux/src/remote_install.rs`

- [ ] **Step 1: Probe remote for existing server**

```rust
async fn probe_remote_server(session: &ssh::Session) -> Result<Option<String>> {
    let output = session.exec("command -v zerminal-server").await?;
    let path = output.trim();
    if path.is_empty() {
        // Also check ~/.zerminal-server/
        let output2 = session.exec("ls ~/.zerminal-server/zerminal-server 2>/dev/null").await?;
        let path2 = output2.trim();
        if path2.is_empty() {
            Ok(None)
        } else {
            Ok(Some(path2.to_string()))
        }
    } else {
        Ok(Some(path.to_string()))
    }
}
```

- [ ] **Step 2: Same-arch install via scp**

```rust
async fn install_same_arch(session: &ssh::Session) -> Result<String> {
    let local_binary = std::env::current_exe()?;  // or known path
    let remote_dir = "~/.zerminal-server";
    session.exec(format!("mkdir -p {}", remote_dir)).await?;
    session.scp(&local_binary, &format!("{}/zerminal-server", remote_dir)).await?;
    session.exec(format!("chmod +x {}/zerminal-server", remote_dir)).await?;
    Ok(format!("{}/zerminal-server", remote_dir))
}
```

- [ ] **Step 3: Cross-arch install via download**

```rust
async fn install_cross_arch(session: &ssh::Session, target_arch: &str) -> Result<String> {
    let url = format!(
        "https://github.com/zerminal/zerminal/releases/latest/download/zerminal-server-{}",
        target_arch
    );
    session.exec(format!(
        "mkdir -p ~/.zerminal-server && curl -L {} -o ~/.zerminal-server/zerminal-server && chmod +x ~/.zerminal-server/zerminal-server",
        url
    )).await?;
    Ok("~/.zerminal-server/zerminal-server".to_string())
}
```

- [ ] **Step 4: Detect remote arch**

```rust
async fn detect_remote_arch(session: &ssh::Session) -> Result<String> {
    let output = session.exec("uname -m").await?;
    Ok(output.trim().to_string())
}
```

pub async fn sync_extensions_to_remote(domain: &MuxDomain) -> Result<()> {
    let local_exts = scan_extensions_dir()?;
    for ext in local_exts {
        if ext.runtime.side == ServerSide || ext.runtime.side == Both {
            if ext.runtime.sync {
                domain.install_remote_extension(&ext.manifest, &ext.source).await?;
            }
        }
    }
    Ok(())
}
